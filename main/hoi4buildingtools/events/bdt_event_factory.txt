# 民用工廠
country_event = {
	id = bdt.event.factory.21200
	title = bdt.event.factory.21200.t
	desc = bdt.event.factory.21200.d

	is_triggered_only = yes

	trigger = {
		is_ai = no
	}

	# 移除idea
	immediate = {
		hidden_effect = { 
			remove_ideas = bdt_industrial_idea 
		}
	}


	# 隨機在2個省各添加1個民用工廠（總共2個民用工廠）
	option = {
		name = bdt.event.factory.21200.a

		random_owned_state = {
			limit = {
				# free_building_slots = {
				# 	include_locked = yes
				# }		
			}

			# 添加工廠
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			} # 添加工廠

		} # random_owned_state

		PREV = {
			random_owned_state = {
				limit = {
					NOT = {
						state = PREV
					}		
				}

				# 添加工廠
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				} # 添加工廠

			} # random_owned_state
		}

	} # 隨機在2個省各添加1個民用工廠

	# 添加5個民用工廠
	option = {
		name = bdt.event.factory.21200.b

	}

	# 關閉
	option = {
		name = bdt.event.common.b
	}

}
