# 民用工廠
country_event = {
	id = bdt.event.factory.21200
	title = bdt.event.factory.21200.t
	desc = bdt.event.factory.21200.d

	is_triggered_only = yes

	trigger = {
		is_ai = no
	}

	# 移除idea
	immediate = {
		hidden_effect = { 
			remove_ideas = bdt_industrial_idea 
		}
	}


	# 每省各添加1個民用工廠
	option = {
		name = bdt.event.factory.21200.a

		every_owned_state = {
			limit = {
				is_core_of = ROOT
				
				# free_building_slots = {
				# 	include_locked = yes
				# }		
			}

			# 添加工廠
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			} # 添加工廠

		} # every_owned_state
		    
	} # 每省各添加1個民用工廠

	# 每省各添加2個民用工廠
	option = {
		name = bdt.event.factory.21200.b

		every_owned_state = {
			limit = {
				is_core_of = ROOT

				# free_building_slots = {
				# 	include_locked = yes
				# }		
			}

			# 每省各添加2個工廠
			add_building_construction = {
				type = industrial_complex
				level = 2
				instant_build = yes
			} # 添加工廠

		} # every_owned_state

	} # 每省各添加2個民用工廠


	# 每省各添加5個民用工廠
	option = {
		name = bdt.event.factory.21200.c

		every_owned_state = {
			limit = {
				is_core_of = ROOT

				# free_building_slots = {
				# 	include_locked = yes
				# }		
			}

			# 每省各添加2個工廠
			add_building_construction = {
				type = industrial_complex
				level = 5
				instant_build = yes
			} # 添加工廠

		} # every_owned_state

	} # 每省各添加5個民用工廠


	# 關閉
	option = {
		name = bdt.event.common.b
	}

}
